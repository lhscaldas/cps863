\section*{Questão 2}

\begin{tikzpicture}[
    node distance=1.5cm and 2cm,
    every node/.style={draw, circle, minimum size=1cm, font=\small},
    every edge/.style={draw, -{Latex}, thick},
]

% Define os estados
\foreach \C in {0,...,8} {
    \foreach \S in {1,2,3} {
        \node (C\C S\S) at (\S*2,-\C*1.5) {(\C, \S)};
    }
}

% Transições para manutenção dos servidores (ação 0)
\foreach \C in {0,...,8} {
    \foreach \S in {1,2,3} {
        \ifnum\C<8
            \path (C\C S\S) edge[bend left=20] node[right] {\(p_0, p_2, p_4\)} (C\the\numexpr\C+1 S\S);
        \fi
    }
}

% Transições para adicionar servidores (ação +1)
\foreach \C in {0,...,8} {
    \foreach \S [count=\i from 1] in {1,2} { % Apenas até S = 2
        \ifnum\C<8
            \path (C\C S\S) edge[bend left=20] node[below] {+1} (C\C S\the\numexpr\S+1);
        \fi
    }
}

% Transições para remover servidores (ação -1)
\foreach \C in {0,...,8} {
    \foreach \S [count=\i from 2] in {2,3} { % Apenas a partir de S = 2
        \path (C\C S\S) edge[bend right=20] node[above] {-1} (C\C S\the\numexpr\S-1);
    }
}

% Legenda para os estados
\node[draw=none, align=left, anchor=north west] at (-1, 1) {
    \textbf{Legenda:} \\
    \((C, S)\): \(C\) = clientes, \(S\) = servidores \\
    \(+1\): adicionar servidor \\
    \(-1\): remover servidor \\
    \(p_0, p_2, p_4\): probabilidades de chegada de clientes
};

\end{tikzpicture}
